from flask import Flask, render_template, request, redirect, url_for, flash, jsonify
from werkzeug.utils import secure_filename
import json
import base64
import os

app = Flask(__name__)
app.secret_key = "your_secret_key_here"

STATUS_FILE = "status.json"
FACULTY_FILE = "faculty_data.json"
TIMETABLE_FILE = "timetable.json"
PHOTOS_FOLDER = "static/faculty_photos"


# Reset status when app starts
def reset_status():
    if os.path.exists(STATUS_FILE):
        with open(STATUS_FILE, "r") as f:
            current = json.load(f)
    else:
        current = [
            {"room": "CSE-A", "status": "Absent", "faculty": ""},
            {"room": "CSE-B", "status": "Absent", "faculty": ""},
            {"room": "CSE-C", "status": "Absent", "faculty": ""}
        ]

    for room in current:
        room["status"] = "Absent"
        room["faculty"] = ""

    with open(STATUS_FILE, "w") as f:
        json.dump(current, f, indent=4)


reset_status()


def load_class_status():
    if os.path.exists(STATUS_FILE):
        with open(STATUS_FILE, "r") as f:
            return json.load(f)
    return []


def load_faculty_data():
    if os.path.exists(FACULTY_FILE):
        with open(FACULTY_FILE, "r") as f:
            return json.load(f)
    return {}


def enrich_class_status(classrooms, faculty_data):
    for room in classrooms:
        faculty_id = room.get("faculty", "")

        if faculty_id and faculty_id in faculty_data:
            details = faculty_data[faculty_id]
            room["name"] = details["name"]
            room["phone"] = details["phone"]
            room["subject"] = details["subject"]

            # Photo path
            photo_filename = details["image"]
            photo_path = os.path.join(PHOTOS_FOLDER, photo_filename)
            if os.path.exists(photo_path):
                room["photo"] = f"/{PHOTOS_FOLDER}/{photo_filename}"
            else:
                room["photo"] = f"/{PHOTOS_FOLDER}/default.jpg"
        else:
            room["name"] = ""
            room["phone"] = ""
            room["subject"] = ""
            room["photo"] = f"/{PHOTOS_FOLDER}/default.jpg"

    return classrooms


# --- Timetable Helpers ---

@app.route("/save_timetable", methods=["POST"])
def save_timetable_route():
    data = load_timetable()
    new_data = {}

    for day, sessions in data.items():
        new_sessions = []
        for idx in range(len(sessions)):
            faculty = request.form.get(f"{day}-{idx}-faculty")
            subject = request.form.get(f"{day}-{idx}-subject")
            start = request.form.get(f"{day}-{idx}-start")
            end = request.form.get(f"{day}-{idx}-end")
            phone = request.form.get(f"{day}-{idx}-phone")

            new_sessions.append({
                "faculty_name": faculty,
                "subject": subject,
                "start_time": start,
                "end_time": end,
                "phone": phone
            })
        new_data[day] = new_sessions

    save_timetable(new_data)
    flash("Timetable updated successfully!")  
    return redirect(url_for("timetable"))

@app.route("/save_faculty", methods=["POST"])
def save_faculty():
    faculty_id = request.form.get("facultyId", "").strip()
    full_name = request.form.get("fullName", "").strip()
    phone = request.form.get("phone", "").strip()
    subject = request.form.get("subject", "").strip()
    captured_img = request.form.get("capturedImg")

    if not faculty_id:
        return jsonify({"message": "Faculty ID is required"}), 400

    if captured_img:
        img_data = captured_img.split(",")[1]
        known_path = os.path.join("known_faces", f"{faculty_id}.png")
        with open(known_path, "wb") as f:
            f.write(base64.b64decode(img_data))
            
    filename = f"{faculty_id}.jpg"
    photo_path = os.path.join(PHOTOS_FOLDER, filename)

    # --- Case 1: Captured webcam image ---
    if captured_img and captured_img.startswith("data:image"):
        try:
            img_data = captured_img.split(",")[1]
            with open(photo_path, "wb") as f:
                f.write(base64.b64decode(img_data))
        except Exception as e:
            return jsonify({"message": f"Error saving captured image: {str(e)}"}), 500

    # --- Case 2: Uploaded file ---
    uploaded_file = request.files.get("uploadedPhoto")
    if uploaded_file and uploaded_file.filename != "":
        uploaded_file.save(photo_path)

    # --- Update JSON ---
    faculty_data = load_faculty_data()
    faculty_data[faculty_id] = {
        "name": full_name,
        "phone": phone,
        "subject": subject,
        "image": filename if os.path.exists(photo_path) else "default.jpg"
    }

    with open(FACULTY_FILE, "w") as f:
        json.dump(faculty_data, f, indent=4)

    return jsonify({"message": f"Faculty {full_name} saved successfully!"})

    return render_template("faculty_database.html", faculty=faculty)

@app.route("/add_faculty", methods=["POST"])
def add_faculty():
    faculty_id = request.form.get("facultyId").strip()
    full_name = request.form.get("fullName").strip()
    phone = request.form.get("phone").strip()
    subject = request.form.get("subject").strip()
    uploaded_file = request.files.get("uploadedPhoto")

    if not faculty_id:
        return jsonify({"message": "Faculty ID is required"}), 400

    # Save uploaded photo as <faculty_id>.jpg
    if uploaded_file:
        filename = secure_filename(f"{faculty_id}.jpg")
        upload_path = os.path.join(PHOTOS_FOLDER, filename)
        uploaded_file.save(upload_path)   # ← This line saves the file
    else:
        filename = "default.jpg"

    # Save to faculty_data.json
    faculty_data = load_faculty_data()
    faculty_data[faculty_id] = {
        "name": full_name,
        "phone": phone,
        "subject": subject,
        "image": filename
    }

    with open(FACULTY_FILE, "w") as f:
        json.dump(faculty_data, f, indent=4)

    return jsonify({"message": f"Faculty {full_name} added successfully!"})



@app.route("/delete_faculty", methods=["POST"])
def delete_faculty():
    data = request.get_json()
    faculty_id = data.get("facultyId")
    if not faculty_id:
        return jsonify({"message": "Faculty ID required"}), 400

    faculty_data = load_faculty_data()
    if faculty_id in faculty_data:
        # Delete photo
        photo_path = os.path.join(PHOTOS_FOLDER, faculty_data[faculty_id]["image"])
        if os.path.exists(photo_path) and faculty_data[faculty_id]["image"] != "default.jpg":
            os.remove(photo_path)
        # Delete from JSON
        del faculty_data[faculty_id]
        with open(FACULTY_FILE, "w") as f:
            json.dump(faculty_data, f, indent=4)
        return jsonify({"message": f"Faculty {faculty_id} deleted successfully!"})
    else:
        return jsonify({"message": "Faculty not found"}), 404

def update_faculty_database_with_photo():
    faculty_data = load_faculty_data()

    faculty_id = request.form.get("facultyId")

    # Update JSON

    return jsonify({"message": f"Faculty {name} updated successfully!"})

    app.run(debug=True)

    updates = request.json

        json.dump(timetable, f, indent=2)

    return jsonify({"status": "success"})

@app.route("/edit/<faculty_id>", methods=["GET", "POST"])
def edit_faculty(faculty_id):
    data = load_faculty()
    faculty = next((f for f in data if f["faculty_id"] == faculty_id), None)

    if not faculty:
        return "Faculty not found", 404

    if request.method == "POST":
        faculty["faculty_name"] = request.form["faculty_name"]
        faculty["subject"] = request.form["subject"]
        faculty["mobile"] = request.form["mobile"]

        image = request.files.get("faculty_image")
        if image and image.filename != "":
            filename = os.path.join(UPLOAD_FOLDER, secure_filename(f"{faculty_id}_{image.filename}"))
            image.save(filename)
            faculty["image"] = filename

        save_faculty(data)
        return redirect(url_for("facultydata"))

    return render_template("edit_faculty.html", faculty=faculty)

    data = load_faculty()

    if not faculty:
        return jsonify({"error": "Faculty not found"}), 404

    update_data = request.json
    faculty["faculty_name"] = update_data.get("faculty_name", faculty["faculty_name"])
    faculty["subject"] = update_data.get("subject", faculty["subject"])
    faculty["mobile"] = update_data.get("mobile", faculty.get("mobile", ""))

    save_faculty(data)
    return jsonify({"status": "success"})
