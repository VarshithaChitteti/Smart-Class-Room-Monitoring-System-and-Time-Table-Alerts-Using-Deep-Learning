<script>
document.addEventListener("DOMContentLoaded", function() {
    const searchBox = document.getElementById("searchBox");
    const tableBody = document.querySelector("#facultyTable tbody");
    const noFacultyMsg = document.getElementById("noFacultyMsg");
    const saveBtn = document.getElementById("saveChangesBtn");
    const addBtn = document.getElementById("addFacultyBtn");

    // --- Search Filter ---
    searchBox.addEventListener("keyup", function() {
        const filter = this.value.toLowerCase();
        const rows = Array.from(tableBody.querySelectorAll("tr"));
        let hasVisible = false;

        rows.forEach(row => {
            const name = row.querySelector('input[data-field="name"]').value.toLowerCase();
            const phone = row.querySelector('input[data-field="phone"]').value.toLowerCase();
            const subject = row.querySelector('input[data-field="subject"]').value.toLowerCase();

            if (name.includes(filter) || phone.includes(filter) || subject.includes(filter)) {
                row.style.display = "";
                hasVisible = true;
            } else {
                row.style.display = "none";
            }
        });

        noFacultyMsg.style.display = hasVisible ? "none" : "";
    });

    // --- Save Changes ---
   saveBtn.addEventListener("click", function() {
    const rows = Array.from(tableBody.querySelectorAll("tr"));
    const requests = [];

    rows.forEach(row => {
        const idInput = row.querySelector('input[data-field="id"]');
        const nameInput = row.querySelector('input[data-field="name"]');
        const phoneInput = row.querySelector('input[data-field="phone"]');
        const subjectInput = row.querySelector('input[data-field="subject"]');
        const photoInput = row.querySelector('input.photo-upload');

        const facultyId = idInput.value.trim();
        if (!facultyId) return; // skip empty rows

        const formData = new FormData();
        formData.append("facultyId", facultyId);
        formData.append("name", nameInput.value.trim());
        formData.append("phone", phoneInput.value.trim());
        formData.append("subject", subjectInput.value.trim());

        if (photoInput && photoInput.files.length > 0) {
            formData.append("uploadedPhoto", photoInput.files[0]);
        }

        // push the fetch promise
        requests.push(
            fetch("/update_faculty_database_with_photo", {
                method: "POST",
                body: formData
            })
            .then(res => {
                if (!res.ok) throw new Error("Server error");
                return res.json();
            })
        );
    });

    Promise.all(requests)
        .then(results => {
            console.log(results); // logs each success message
            alert("✅ All changes saved successfully!");
        })
        .catch(err => {
            console.error(err);
            alert("❌ Error saving some changes. Check console for details.");
        });
});



    // --- Delete Faculty ---
    tableBody.addEventListener("click", function(e) {
        if (e.target.classList.contains("delete-btn")) {
            const fid = e.target.dataset.id;
            if (confirm("Are you sure you want to delete this faculty?")) {
                fetch("/delete_faculty", {
                    method: "POST",
                    headers: { "Content-Type": "application/json" },
                    body: JSON.stringify({ facultyId: fid })
                })
                .then(res => res.json())
                .then(data => {
                    alert(data.message);
                    e.target.closest("tr").remove();
                });
            }
        }
    });

    // --- Add New Faculty ---
    document.getElementById("addFacultyBtn").addEventListener("click", function() {
    const tableBody = document.querySelector("#facultyTable tbody");

    const newRow = document.createElement("tr");

    newRow.innerHTML = `
        <td><input type="text" class="editable" data-field="id" data-id=""></td> <!-- Faculty ID -->
        <td>
            <img src="{{ url_for('static', filename='faculty_photos/default.jpg') }}" class="faculty-photo" alt="photo">
            <input type="file" class="photo-upload">
        </td>
        <td><input type="text" class="editable" data-field="name" data-id=""></td> <!-- Name -->
        <td><input type="text" class="editable" data-field="phone" data-id=""></td> <!-- Phone -->
        <td><input type="text" class="editable" data-field="subject" data-id=""></td> <!-- Subject -->
        <td><button class="delete-btn">Delete</button></td> <!-- Actions -->
    `;

    tableBody.appendChild(newRow);
});

});
</script>

    function searchFaculty() {
      let input = document.getElementById("searchInput").value.toLowerCase();
      let table = document.getElementById("facultyTable");
      let rows = table.getElementsByTagName("tr");

        let cells = rows[i].getElementsByTagName("td");
        let match = false;

        for (let j = 0; j < cells.length; j++) {
          if (cells[j].innerText.toLowerCase().includes(input)) {
            match = true;
            break;
          }
        }
        rows[i].style.display = match ? "" : "none";
      }
    }
